// function to detect how many decimal places
let decPl = function () {
    function isInt(n) {
       return typeof n === 'number' && parseFloat(n) == parseInt(n, 10) && !isNaN(n);
    }
    return function (n) {
       const a = Math.abs(n);
       let c = a, count = 1;
       while (!isInt(c) && isFinite(c)) {
          c = a * Math.pow(10, count++);
       }
       return count - 1;
    };
 }();

// function generates a random number from a range of 2 inputs
function randge (lower = 0, higher = 1) {
    if (lower % 1 === 0 && higher % 1 === 0) {
        return (lower + Math.round((higher - lower) * Math.random()))
    } else {
        return (lower + ((Math.round((Math.pow(10, decPl(lower)) * (higher - lower)) * Math.random()) / Math.pow(10, decPl(lower)))))
    }
}
